\section{Why?}
\begin{frame}
	\frametitle{Why? For taking considered decisions}
	
	Help \ac{DM} take considered decisions
	\begin{itemize}
		\item Which site for a nuclear plant?
		\item Which smartphone should I buy?
		\item Which voting rule should we implement?
	\end{itemize}
	My main interest is in analyzing such situations abstractly and generally
\end{frame}

\section{What?}
\begin{frame}
	\frametitle{What? Deliberated preferences}
	\begin{block}{Intuitive preference}
		\begin{itemize}
			\item Preference as an “immediate sensation” \citep{von_neumann_theory_1944}
			\item $i$ knows what’s best by introspection
			\item Recommend a movie: $i$ knows how good it feels
			\item “There is, of course, an important sense in which preferences, being entirely subjective, cannot be in error” \citep{savage_foundations_1972}
		\end{itemize}
	\end{block}
	\begin{block}{Deliberated preference}
		\begin{itemize}
			\item … “but in a different, more subtle sense they can be.”
			\item Learn (think) more then change your mind
			\item Relates to “slow thinking” \citep{kahneman_thinking_2013}
		\end{itemize}
	\end{block}
	I want to define precisely and analyze deliberated preferences
\end{frame}

\section{How?}
\begin{frame}
	\frametitle{How? Axiomatic analysis and argumentation}
	\vspace{-2pt}
	\begin{center}
		\begin{tikzpicture}
			\path node[matrix of nodes, ampersand replacement=\&, inner sep=0, nodes={inner sep=.3333em}] (my matrix) {
					\& aspect 1	\& aspect 2	\& aspect 3	\& aspect 4\\
				$a$	\& 7	\& 10	\& 7	\& | (st 1) | −\\
				$b$	\& 1.5	\& 5	\& 1	\& | (st 2) | −−\\
				$c$	\& 7.5	\& 2	\& 5	\& | (st 3) | +\\
				$d$	\& 2.5	\& 3	\& 3	\& | (st 4) | ++\\
			};
			\path (st 2.south -| my matrix.east) coordinate (start);
			\path (start) ++ (20mm, 0) node[anchor=west] (end) {$a > b$};
			\draw[->] (start) -- (end);
			\path (end.west) ++(-10mm, 0.3cm) node (label) {$f$};
		\end{tikzpicture}
	\end{center}
	\begin{itemize}
		\item A family of aggregators $\mathcal{F}$
		\item $P_1, P_2$: properties of aggregators (Pareto-Dominance, Sure thing principle, Condorcet…)
		\item Explain to the \ac{DM}: if $P$, then $a > b$
		\item[→] Argue: $a > b$, otherwise $¬P$
	\end{itemize}
	\begin{block}{Enables}
		\begin{itemize}
			\item Test axioms
			\item Compare aggregators empirically
		\end{itemize}
	\end{block}
\end{frame}
